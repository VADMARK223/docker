`WaitGroup` –ø–æ–∑–≤–æ–ª—è–µ—Ç –¥–æ–∂–¥–∞—Ç—å—Å—è –∑–∞–≤–µ—Ä—à–µ–Ω–∏—è –≥—Ä—É–ø–ø—ã –≥–æ—Ä—É—Ç–∏–Ω. –°–Ω–∞—á–∞–ª–∞ –≤—ã–∑—ã–≤–∞–µ–º `Add(n)` –¥–ª—è —á–∏—Å–ª–∞ –≥–æ—Ä—É—Ç–∏–Ω, –∫–∞–∂–¥–∞—è –≤—ã–∑—ã–≤–∞–µ—Ç `Done()` –ø–æ –∑–∞–≤–µ—Ä—à–µ–Ω–∏—é, –∞ `Wait()` –±–ª–æ–∫–∏—Ä—É–µ—Ç –≤—ã–ø–æ–ª–Ω–µ–Ω–∏–µ, –ø–æ–∫–∞ –≤—Å–µ –≥–æ—Ä—É—Ç–∏–Ω—ã –Ω–µ –∑–∞–≤–µ—Ä—à–∞—Ç—å—Å—è. –≠—Ç–æ –æ—á–µ–Ω—å —É–¥–æ–±–Ω–æ –¥–ª—è —Å–∏–Ω—Ö—Ä–æ–Ω–∏–∑–∞—Ü–∏–∏ –ø–∞—Ä–∞–ª–ª–µ–ª—å–Ω—ã—Ö –∑–∞–¥–∞—á.

    wg.Add(1)
    go func(id int) {

        for j := 0; j < 3; j++ {
            counter++ // ‚ùå –≥–æ–Ω–∫–∞!
